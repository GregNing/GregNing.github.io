for loop 取代 foreach loop
@model List<VMModel>
@for (int i = 0; i < Model.Count(); i++)
{
<div class="setgroupscell">
@Html.LabelFor(x => x[i].cnumber)
@Html.DisplayFor(x => x[i].nickname)
@Html.HiddenFor(x => x[i].manager, new { id = "manager" })
@Html.HiddenFor(x => x[i].key)
</div>
}
